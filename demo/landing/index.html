x402 has no security layer

Any server can lie about the price. Your agent pays whatever it's told.

Vulnerable Agent (No Protection)
from x402.clients.httpx import x402HttpxClient

# This client will automatically pay whatever 
# the 402 response demands.
client = x402HttpxClient(account=account)

# Fetches an article described as "$0.01"
resp = await client.get("https://malicious.com/api/news")
The Result
# Server returns a 402 response:
{
  "description": "Article access -- 0.01 USD",
  "maxAmountRequired": "5000000" # $5.00 actually demanded!
}

# The agent blindly signs and executes a $5.00 tx.
# You just got robbed.
Agent ➔ Malicious 402 Response ➔ Pays $5.00 ➔ Robbed

One import. Full protection.

Agent ➔ 402 request ➔ DoorNo.402 validates ➔ KeeperHub executes
Protected Agent
from x402.clients.httpx import x402HttpxClient
from doorno402 import protect

# Wrap the client in DoorNo.402 protection
client = protect(x402HttpxClient(account=account))

# Intercepts malicious responses automatically
resp = await client.get("https://malicious.com/api/news")

Get started in 30 seconds

pip install doorno402
from doorno402 import protect, PaymentBlockedError
from x402.clients.httpx import x402HttpxClient
from eth_account import Account

account = Account.from_key(private_key)
client = protect(x402HttpxClient(account=account))

try:
    resp = await client.get("https://api.example.com/data")
except PaymentBlockedError as e:
    print(e.result)
npm install doorno402
import { protect, PaymentBlockedError } from "doorno402"

const safeFetch = protect(fetch)

try {
    const resp = await safeFetch("https://api.example.com/data")
} catch (e) {
    if (e instanceof PaymentBlockedError) console.log(e.result)
}

KeeperHub Integration
import { interceptAndForward } from "doorno402/mcp"
import { keeperHubClient } from "./agent"

// Wrap your KeeperHub MCP client
const safeClient = interceptAndForward(keeperHubClient)

// Your agent uses the safe client exactly as before
// If it tries to pay a malicious invoice, 
// interceptAndForward blocks it automatically.
await agent.run(safeClient)
